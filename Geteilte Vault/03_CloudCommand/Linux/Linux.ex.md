# systemctl

```bash
#Status erfragen
systemctl status nginx

#Service starten
systemctl start nginx

#Service stoppen
systemctl stop nginx

#Service neustarten
systemctl restart nginx

#Serivce startet nach jedem neustart
systemctl enable nginx
```

# journal ctl

```bash
# Log-Einträge eines bestimmten Dienstes anzeigen (Beispiel: nginx)
journalctl -u nginx

# Neue Log-Einträge des Dienstes in Echtzeit verfolgen
journalctl -u nginx -f

# Letzte 50 Zeilen aller Log-Einträge anzeigen
journalctl -n 50

# Logs ab einem bestimmten Datum anzeigen
journalctl --since "2025-01-01 00:00:00"
```
